## Parenthesis

Parenthesis matching is a fundamental stack problem.

**Simple case** - just count open/close (works for single type of bracket):

```python
def isValid(self, s) -> bool:
  cnt = 0
  for c in s:
    if c == '(':
      cnt += 1             # open bracket: increment depth
    elif c == ')':
      cnt -= 1             # close bracket: decrement depth
      if cnt < 0:          # more closes than opens = invalid!
        return False
  return cnt == 0          # must have equal opens and closes
```

**Multiple bracket types** - use a stack:

```python
def isValid(self, s: str) -> bool:
  stk = []
  mp = {")":"(", "}":"{", "]":"["}   # maps closing -> expected opening
    for c in s:
      if c in mp.values():           # opening bracket
        stk.append(c)               # push onto stack
      elif c in mp.keys():          # closing bracket
        test = stk.pop() if stk else '#'  # pop top (or '#' if empty)
        if mp[c] != test:           # doesn't match expected opening
          return False
  return len(stk) == 0              # stack should be empty if all matched
```

**Track indices for removal** - when you need to know WHICH brackets to remove:

```python
def minRemoveToMakeValid(self, s: str) -> str:
  rtn = list(s)                    # convert string to list (strings are immutable)
  stk = []                         # stack of indices of unmatched '('
  for i, c in enumerate(s):
    if c == '(':
      stk.append(i)               # record index of open bracket
    elif c == ')':
      if len(stk) > 0:
        stk.pop()                  # matched with an open bracket
      else:
        rtn[i] = ''               # unmatched close bracket: mark for removal
  while stk:
    rtn[stk.pop()] = ''           # remaining unmatched open brackets: mark for removal
  return "".join(rtn)
```
