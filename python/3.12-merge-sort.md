## Merge Sort

Merge sort divides the array in half, recursively sorts each half, then merges the two sorted halves. Always O(n log n) but uses O(n) extra space.

```python
from collections import deque
def mergeSort(array):
    def sortArray(nums):
        if len(nums) > 1:              # base case: single element is sorted
            mid = len(nums)//2
            l1 = sortArray(nums[:mid])  # sort left half
            l2 = sortArray(nums[mid:])  # sort right half
            nums = sort(l1, l2)        # merge two sorted halves
        return nums

    def sort(l1, l2):
        result = []
        l1 = deque(l1)                 # deque for O(1) popleft
        l2 = deque(l2)
        while l1 and l2:               # while both have elements
            if l1[0] <= l2[0]:         # take smaller of the two fronts
                result.append(l1.popleft())
            else:
                result.append(l2.popleft())
        result.extend(l1 or l2)        # append remaining elements (one list is empty)
        return result
    return sortArray(array)
```
