## Reverse Polish

Reverse Polish Notation (RPN) evaluates expressions without parentheses. Numbers go on stack; operators pop two numbers, compute, push result.

```python
class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stk = []
        while tokens:
            c = tokens.pop(0)          # get next token
            if c not in '+-/*':        # it's a number
                stk.append(int(c))
            else:                      # it's an operator
                a = stk.pop()          # second operand (top of stack)
                b = stk.pop()          # first operand (below top)
                # Note: order matters! b op a (b was pushed first)
                if c == '+':
                    stk.append(a + b)
                if c == '-':
                    stk.append(b - a)  # b - a (not a - b!)
                if c == '*':
                    stk.append(a * b)
                if c == '/':
                    stk.append(int(b / a))  # int() truncates toward zero
        return stk[0]                  # final answer is only element left
```
