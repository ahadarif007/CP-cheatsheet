## Tree Tricks

**Pass min/max down to leaves** - track running min/max as you traverse:

```
         8                 path: 8→3→1
        / \                minval=1, maxval=8
       3   10              ans = |8-1| = 7
      / \    \
     1   6   14            each root-to-leaf path tracks its own min/max
```

```python
def maxAncestorDiff(self, root: TreeNode) -> int:
    if not root:
        return 0
    self.ans = 0
    def dfs(node, minval, maxval):
        if not node:
            self.ans = max(self.ans, abs(maxval - minval))
            return
        dfs(node.left, min(node.val, minval), max(node.val, maxval))
        dfs(node.right, min(node.val, minval), max(node.val, maxval))
    dfs(root, float('inf'), float('-inf'))
    return self.ans
```

**Build paths as strings** - store path-so-far in the stack alongside each node:

```python
def binaryTreePaths(self, root: TreeNode) -> List[str]:
    if root is None: return []
    rtn = []
    stk = [(root, str(root.val))]          # (node, path_string_so_far)
    while stk:
        node, path = stk.pop()
        if not node.left and not node.right:   # leaf node
            rtn.append(path)
        if node.left:
            stk.append((node.left, path + "->" + str(node.left.val)))
        if node.right:
            stk.append((node.right, path + "->" + str(node.right.val)))
    return rtn
```

**Return value = height, global = answer** - DFS where return value tracks one thing, global tracks another:

```
Diameter = longest path between any two nodes (in edges)

         1
        / \
       2   3          left height = 2, right height = 1
      / \             diameter through node 1 = 2 + 1 = 3
     4   5
                      return value: height (for parent's calculation)
                      global variable: best diameter seen so far
```

```python
def diameterOfBinaryTree(self, root: TreeNode) -> int:
    self.mx = 0
    def dfs(node):
        if not node:
            return 0
        l = dfs(node.left)            # height of left subtree
        r = dfs(node.right)           # height of right subtree
        self.mx = max(self.mx, l + r) # diameter through this node
        return max(l, r) + 1          # return HEIGHT (not diameter)
    dfs(root)
    return self.mx
```

**Convert tree to graph** - when you need to find nodes at distance K or traverse "upward":

```
  Tree:       3              Graph (bidirectional):
             / \               3 ↔ 5, 3 ↔ 1
            5   1              5 ↔ 6, 5 ↔ 2
           / \                 1 ↔ 0, 1 ↔ 8
          6   2
             / \             Now you can BFS/DFS from any node
            7   4            in any direction (including "up"!)
```

```python
def distanceK(self, root: TreeNode, target: TreeNode, K: int) -> List[int]:
    adj = collections.defaultdict(list)

    def build(node):
        if node.left:
            adj[node].append(node.left)
            adj[node.left].append(node)      # child → parent (key insight!)
            build(node.left)
        if node.right:
            adj[node].append(node.right)
            adj[node.right].append(node)
            build(node.right)
    build(root)

    def dfs(node, prev, d):
        if node:
            if d == K:
                rtn.append(node.val)
            else:
                for nei in adj[node]:
                    if nei != prev:          # don't revisit parent
                        dfs(nei, node, d + 1)

    rtn = []
    dfs(target, None, 0)
    return rtn
```
