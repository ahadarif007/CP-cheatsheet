## Tree Tricks

**Pass min/max down to leaves** - track running min/max as you traverse:

```python
def maxAncestorDiff(self, root: TreeNode) -> int:
    if not root:
        return 0
    self.ans = 0
    def dfs(node, minval, maxval):
        if not node:
            # At leaf: the max difference on this root-to-leaf path
            self.ans = max(self.ans, abs(maxval - minval))
            return
        # Update min/max with current node's value, pass down to children
        dfs(node.left, min(node.val, minval), max(node.val, maxval))
        dfs(node.right, min(node.val, minval), max(node.val, maxval))
    dfs(root, float('inf'), float('-inf'))
    return self.ans
```

**Build paths as strings** - store path-so-far in the stack alongside each node:

```python
def binaryTreePaths(self, root: TreeNode) -> List[str]:
    rtn = []
    if root is None: return []
    stk = [(root, str(root.val))]          # (node, path_string_so_far)
    while stk:
        node, path = stk.pop()
        if node.left is None and node.right is None:  # leaf node
            rtn.append(path)               # complete path found
        if node.left:
            stk.append((node.left, path + "->" + str(node.left.val)))
        if node.right:
            stk.append((node.right, path + "->" + str(node.right.val)))
    return rtn
```

**Return value = height, global variable = answer** - a common DFS pattern where the return value tracks one thing and a global tracks another:

```python
def diameterOfBinaryTree(self, root: TreeNode) -> int:
    self.mx = 0                   # global max diameter found
    def dfs(node):
        if node:
            l = dfs(node.left)    # height of left subtree
            r = dfs(node.right)   # height of right subtree
            total = l + r         # diameter passing through this node
            self.mx = max(self.mx, total)  # update global max
            return max(l, r) + 1  # return HEIGHT (longest single path down)
        else:
            return 0              # null node has height 0
    dfs(root)
    return self.mx
```

**Convert tree to graph** - when you need to find nodes at distance K or traverse "upward":

```python
def distanceK(self, root: TreeNode, target: TreeNode, K: int) -> List[int]:
    adj = collections.defaultdict(list)  # build bidirectional adjacency list

    def dfsa(node):
        if node.left:
            adj[node].append(node.left)      # parent -> child
            adj[node.left].append(node)      # child -> parent (the key insight!)
            dfsa(node.left)
        if node.right:
            adj[node].append(node.right)
            adj[node.right].append(node)
            dfsa(node.right)
    dfsa(root)

    # Now BFS/DFS from target, treating tree as undirected graph
    def dfs(node, prev, d):
        if node:
            if d == K:
                rtn.append(node.val)    # found node at distance K
            else:
                for nei in adj[node]:
                    if nei != prev:     # don't go back where we came from
                        dfs(nei, node, d+1)

    rtn = []
    dfs(target, None, 0)
    return rtn
```
