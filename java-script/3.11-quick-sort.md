# Quick Sort

```js
function quickSort(a) {
  sort(a, 0, a.length - 1);
}
function sort(a, l, r) {
  if (l >= r) return;
  const p = part(a, l, r);
  sort(a, l, p - 1);
  sort(a, p + 1, r);
}
function part(a, l, r) {
  const pivot = a[r];
  let i = l;
  for (let k = l; k < r; k++)
    if (a[k] < pivot) {
      [a[k], a[i]] = [a[i], a[k]];
      i++;
    }
  [a[r], a[i]] = [a[i], a[r]];
  return i;
}
```
