# Union Find / DSU

```js
class DSU {
  constructor(n) {
    this.p = Array(n)
      .fill(0)
      .map((_, i) => i);
    this.sz = Array(n).fill(1);
  }
  find(x) {
    if (this.p[x] !== x) this.p[x] = this.find(this.p[x]);
    return this.p[x];
  }
  union(a, b) {
    let x = this.find(a),
      y = this.find(b);
    if (x === y) return false;
    if (this.sz[x] < this.sz[y]) [x, y] = [y, x];
    this.p[y] = x;
    this.sz[x] += this.sz[y];
    return true;
  }
}
```
