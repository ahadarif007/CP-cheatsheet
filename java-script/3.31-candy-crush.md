Remove duplicates

```js
function removeDuplicates(s, k) {
  const st = [];
  for (const c of s) {
    if (st.length && st[st.length - 1][0] === c) {
      st[st.length - 1][1]++;
      if (st[st.length - 1][1] === k) st.pop();
    } else st.push([c, 1]);
  }
  let ans = '';
  for (const [ch, cnt] of st) ans += ch.repeat(cnt);
  return ans;
}
```
