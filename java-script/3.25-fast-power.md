# Fast Power

```js
function myPow(x, n) {
  let e = BigInt(n),
    xb = Number.isInteger(x) ? BigInt(x) : x;
  if (e < 0n) {
    xb = 1 / Number(xb);
    e = -e;
  }
  let ans = 1;
  let base = Number(xb);
  while (e > 0n) {
    if ((e & 1n) === 1n) ans *= base;
    base *= base;
    e >>= 1n;
  }
  return ans;
}
```
