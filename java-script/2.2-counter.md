Top-K with counter

```js
function topK(nums, k) {
  const m = new Map();
  for (const x of nums) m.set(x, (m.get(x) || 0) + 1);
  return Array.from(m.entries())
    .sort((a, b) => b[1] - a[1])
    .slice(0, k)
    .map((e) => e[0]);
}
```
