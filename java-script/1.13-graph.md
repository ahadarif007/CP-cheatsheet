# Graph

Adjacency list and BFS/DFS

```js
const g = Array.from({ length: N }, () => []);
for (const [u, v] of edges) {
  g[u].push(v);
  g[v].push(u);
}

function dfs(u, p) {
  for (const v of g[u]) if (v !== p) dfs(v, u);
}

function bfs(s) {
  const order = [],
    vis = Array(N).fill(false);
  const q = [s];
  vis[s] = true;
  for (let qi = 0; qi < q.length; qi++) {
    const u = q[qi];
    order.push(u);
    for (const v of g[u])
      if (!vis[v]) {
        vis[v] = true;
        q.push(v);
      }
  }
  return order;
}
```
