# Parenthesis

> Parenthesis problems use a counter or stack to track open/close balance.

## Count-Only Validation (single type)

Track open count. If it goes negative, there's an unmatched `)`.

```js
function isValidParens(text) {
  let openCount = 0;
  for (const char of text) {
    if (char === '(') openCount++;
    else if (char === ')') {
      openCount--;
      if (openCount < 0) return false;    // more ')' than '('
    }
  }
  return openCount === 0;                 // all '(' matched
}
```

## Stack-Based Validation (multiple bracket types)

```
Input: "{[()]}"
Stack trace:
  '{' -> push      stack: ['{']
  '[' -> push      stack: ['{', '[']
  '(' -> push      stack: ['{', '[', '(']
  ')' -> matches ( stack: ['{', '[']
  ']' -> matches [ stack: ['{']
  '}' -> matches { stack: []
Result: valid (stack empty)
```

```js
function isValidBrackets(text) {
  const stack   = [];
  const matches = new Map([[')', '('], ['}', '{'], [']', '[']]);

  for (const char of text) {
    if ([...matches.values()].includes(char)) {
      stack.push(char);                        // opening bracket: push
    } else if (matches.has(char)) {
      if (!stack.length || stack.pop() !== matches.get(char)) return false;
    }
  }
  return stack.length === 0;
}
```

## Remove Minimum Parentheses to Make Valid

Mark unmatched parentheses for removal, then rebuild the string.

```js
function minRemoveToMakeValid(text) {
  const stack   = [];                          // tracks indices of unmatched '('
  const toRemove = Array(text.length).fill(false);

  for (let i = 0; i < text.length; i++) {
    const char = text[i];
    if (char === '(') {
      stack.push(i);                           // might be unmatched
    } else if (char === ')') {
      if (stack.length) stack.pop();           // matched with a '('
      else toRemove[i] = true;                // unmatched ')'
    }
  }
  while (stack.length) toRemove[stack.pop()] = true;  // remaining unmatched '('

  let result = '';
  for (let i = 0; i < text.length; i++) {
    if (!toRemove[i]) result += text[i];
  }
  return result;
}
```
