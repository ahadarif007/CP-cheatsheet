# Binary Search Tree

Complete tree check by index

```js
function isComplete(root) {
  let total = 0n,
    mx = 0n;
  (function dfs(n, idx) {
    if (!n) return;
    total++;
    if (idx > mx) mx = idx;
    dfs(n.left, idx * 2n);
    dfs(n.right, idx * 2n + 1n);
  })(root, 1n);
  return total === mx;
}
```

Range sum

```js
function rangeSumBST(root, L, R) {
  if (!root) return 0;
  let s = 0;
  const dfs = (n) => {
    if (!n) return;
    if (L <= n.val && n.val <= R) s += n.val;
    if (n.val > L) dfs(n.left);
    if (n.val < R) dfs(n.right);
  };
  dfs(root);
  return s;
}
```

Validate BST (iterative)

```js
function isValidBST(root) {
  const st = [];
  let cur = root,
    prev = null;
  while (cur || st.length) {
    while (cur) {
      st.push(cur);
      cur = cur.left;
    }
    cur = st.pop();
    if (prev !== null && cur.val <= prev) return false;
    prev = cur.val;
    cur = cur.right;
  }
  return true;
}
```
