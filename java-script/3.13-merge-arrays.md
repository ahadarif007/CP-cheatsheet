# Merge K Sorted Arrays

```js
function mergeKSortedArrays(arrays) {
  const pq = new MinHeap((a, b) => a[0] - b[0]);
  for (let i = 0; i < arrays.length; i++)
    if (arrays[i].length > 0) pq.push([arrays[i][0], i, 0]);
  const res = [];
  while (pq.size()) {
    const [val, i, j] = pq.pop();
    res.push(val);
    if (j + 1 < arrays[i].length) pq.push([arrays[i][j + 1], i, j + 1]);
  }
  return res;
}
```
