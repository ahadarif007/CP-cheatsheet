# Topological Sort (Kahn's Algorithm / BFS)

> Topological sort orders nodes in a directed acyclic graph (DAG) so that
> every edge u -> v has u before v. Uses BFS with in-degree counting.
> If the result has fewer nodes than the graph, there's a cycle.

## Course Schedule (Can Finish?)

```
Courses: 0, 1, 2, 3
Prerequisites: [[1,0], [2,0], [3,1], [3,2]]
  Means: 0 -> 1, 0 -> 2, 1 -> 3, 2 -> 3

  0 --> 1 --\
  |          v
  +-------> 3
  |          ^
  0 --> 2 --/

In-degree: [0:0, 1:1, 2:1, 3:2]
BFS: start with 0 (degree 0), process 1, 2, then 3
Result: [0, 1, 2, 3] -> length 4 == n, so CAN finish
```

```js
function canFinish(numCourses, prerequisites) {
  const adjList  = Array.from({ length: numCourses }, () => []);
  const inDegree = Array(numCourses).fill(0);

  for (const [course, prereq] of prerequisites) {
    adjList[prereq].push(course);    // prereq -> course
    inDegree[course]++;
  }

  // Start with all courses that have no prerequisites
  const queue = [];
  for (let i = 0; i < numCourses; i++) {
    if (inDegree[i] === 0) queue.push(i);
  }

  // Process queue: for each completed course, reduce dependents' in-degree
  for (let qi = 0; qi < queue.length; qi++) {
    const course = queue[qi];
    for (const next of adjList[course]) {
      if (--inDegree[next] === 0) queue.push(next);
    }
  }

  return queue.length === numCourses;   // all courses reachable = no cycle
}
```

## Alien Dictionary

Given sorted words in an alien language, determine the character ordering.

```js
function alienOrder(words) {
  const chars   = new Set(words.join(''));
  const adjList = new Map();
  const inDeg   = new Map();

  for (const ch of chars) {
    adjList.set(ch, []);
    inDeg.set(ch, 0);
  }

  // Compare adjacent words to find ordering constraints
  for (let i = 0; i + 1 < words.length; i++) {
    const wordA = words[i], wordB = words[i + 1];
    const minLen = Math.min(wordA.length, wordB.length);
    let foundDiff = false;

    for (let j = 0; j < minLen; j++) {
      if (wordA[j] !== wordB[j]) {
        adjList.get(wordA[j]).push(wordB[j]);   // wordA[j] comes before wordB[j]
        inDeg.set(wordB[j], inDeg.get(wordB[j]) + 1);
        foundDiff = true;
        break;
      }
    }
    if (!foundDiff && wordA.length > wordB.length) return ''; // invalid: "abc" before "ab"
  }

  // BFS topological sort
  const queue = [...Array.from(inDeg.entries())
    .filter(([, deg]) => deg === 0)
    .map(([ch]) => ch)];

  let result = '';
  for (let qi = 0; qi < queue.length; qi++) {
    const ch = queue[qi];
    result += ch;
    for (const next of adjList.get(ch)) {
      inDeg.set(next, inDeg.get(next) - 1);
      if (inDeg.get(next) === 0) queue.push(next);
    }
  }

  return result.length === chars.size ? result : '';  // empty if cycle detected
}
```
