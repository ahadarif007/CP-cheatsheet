# Grids

```js
const R = grid.length,
  C = grid[0].length;
const dirs = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];
function dfs(r, c) {
  grid[r][c] = 2;
  for (const [dr, dc] of dirs) {
    const nr = r + dr,
      nc = c + dc;
    if (0 <= nr && nr < R && 0 <= nc && nc < C && grid[nr][nc] === 1)
      dfs(nr, nc);
  }
}
```
