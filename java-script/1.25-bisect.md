# Bisect

```js
function lowerBound(a, x) {
  let l = 0,
    r = a.length;
  while (l < r) {
    const m = (l + r) >> 1;
    if (a[m] < x) l = m + 1;
    else r = m;
  }
  return l;
}
function upperBound(a, x) {
  let l = 0,
    r = a.length;
  while (l < r) {
    const m = (l + r) >> 1;
    if (a[m] <= x) l = m + 1;
    else r = m;
  }
  return l;
}
```
