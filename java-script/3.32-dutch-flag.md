# Dutch Flag

```js
function sortColors(a) {
  let p0 = 0,
    cur = 0,
    p2 = a.length - 1;
  while (cur <= p2) {
    if (a[cur] === 0) {
      [a[p0], a[cur]] = [a[cur], a[p0]];
      p0++;
      cur++;
    } else if (a[cur] === 2) {
      [a[cur], a[p2]] = [a[p2], a[cur]];
      p2--;
    } else cur++;
  }
}
```
