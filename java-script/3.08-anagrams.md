# Anagrams

```js
function isAnagram(s, t) {
  if (s.length !== t.length) return false;
  const c = Array(26).fill(0);
  for (const ch of s) c[ch.charCodeAt(0) - 97]++;
  for (const ch of t) if (--c[ch.charCodeAt(0) - 97] < 0) return false;
  return true;
}
```

Find all anagram indices (sliding window)

```js
function findAnagrams(s, p) {
  const need = Array(26).fill(0);
  for (const ch of p) need[ch.charCodeAt(0) - 97]++;
  const have = Array(26).fill(0);
  const ans = [];
  const m = p.length;
  for (let i = 0; i < s.length; i++) {
    have[s.charCodeAt(i) - 97]++;
    if (i >= m) have[s.charCodeAt(i - m) - 97]--;
    if (need.every((v, j) => v === have[j])) ans.push(i - m + 1);
  }
  return ans;
}
```
