# String Subsequence (Shortest Way)

> Find the minimum number of subsequences of `source` needed to form `target`.
> Precompute character positions in source for O(1) jumps via binary search.

## How It Works

```
source = "abc", target = "aabbc"

Pass 1: match 'a' at 0, 'a'? no more 'a' after 0... new pass needed
  Actually: 'a'(0) -> 'b'(1) -> need another pass
Pass 2: 'b'(1) -> 'c'(2)

Wait, let me trace properly:
  target chars: a, a, b, b, c
  Pass 1: a(0), b(1), c(2)  -> matched "abc" from "aabbc"
  remaining: a, b
  Pass 2: a(0), b(1)        -> matched "ab"
Answer: 2 passes
```

```js
function shortestWay(source, target) {
  // Precompute: for each character, store sorted list of positions in source
  const charPositions = new Map();
  for (let i = 0; i < source.length; i++) {
    const char = source[i];
    if (!charPositions.has(char)) charPositions.set(char, []);
    charPositions.get(char).push(i);
  }

  let passCount    = 1;
  let sourceIdx    = -1;      // current position in source (exclusive lower bound)

  for (const char of target) {
    if (!charPositions.has(char)) return -1;    // char not in source at all

    const positions = charPositions.get(char);
    let nextPos     = lowerBound(positions, sourceIdx);

    if (nextPos === positions.length) {
      passCount++;                               // start new pass through source
      sourceIdx = positions[0] + 1;
    } else {
      sourceIdx = positions[nextPos] + 1;
    }
  }
  return passCount;

  function lowerBound(arr, target) {
    let left = 0, right = arr.length;
    while (left < right) {
      const mid = (left + right) >> 1;
      if (arr[mid] < target) left = mid + 1;
      else right = mid;
    }
    return left;
  }
}
```
