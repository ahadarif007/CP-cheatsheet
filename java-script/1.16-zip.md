Rotate matrix using transpose+reverse

```js
function rotate(m) {
  const n = m.length;
  for (let i = 0; i < n; i++)
    for (let j = i + 1; j < n; j++) {
      const t = m[i][j];
      m[i][j] = m[j][i];
      m[j][i] = t;
    }
  for (let i = 0; i < n; i++) m[i].reverse();
}
```
