# Tree Tricks

Max ancestor diff

```js
function maxAncestorDiff(root) {
  let ans = 0;
  function dfs(n, mn, mx) {
    if (!n) {
      ans = Math.max(ans, Math.abs(mx - mn));
      return;
    }
    dfs(n.left, Math.min(mn, n.val), Math.max(mx, n.val));
    dfs(n.right, Math.min(mn, n.val), Math.max(mx, n.val));
  }
  dfs(root, Infinity, -Infinity);
  return ans;
}
```

Diameter

```js
function diameterOfBinaryTree(root) {
  let best = 0;
  function depth(n) {
    if (!n) return 0;
    const l = depth(n.left),
      r = depth(n.right);
    best = Math.max(best, l + r);
    return Math.max(l, r) + 1;
  }
  depth(root);
  return best;
}
```
