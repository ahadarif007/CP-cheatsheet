Manual split by isLetter

```js
function splitWords(input) {
  const out = [];
  let start = -1;
  for (let i = 0; i < input.length; i++) {
    const ch = input[i];
    if (/[A-Za-z]/.test(ch)) {
      if (start === -1) start = i;
    } else {
      if (start !== -1) {
        out.push(input.slice(start, i));
        start = -1;
      }
    }
  }
  if (start !== -1) out.push(input.slice(start));
  return out;
}
```
