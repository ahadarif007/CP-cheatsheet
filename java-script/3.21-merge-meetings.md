# Merge Meetings

```js
function insert(intervals, newI) {
  const li = intervals.slice();
  li.push(newI);
  li.sort((a, b) => a[0] - b[0]);
  const res = [li[0]];
  for (const cur of li) {
    const last = res[res.length - 1];
    if (cur[0] <= last[1]) last[1] = Math.max(last[1], cur[1]);
    else res.push(cur);
  }
  return res;
}
```
