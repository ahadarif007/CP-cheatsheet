# Events

```js
function employeeFreeTime(schedule) {
  const ev = [];
  for (const e of schedule)
    for (const m of e) {
      ev.push([m.start, 1]);
      ev.push([m.end, -1]);
    }
  ev.sort((a, b) => a[0] - b[0]);
  const itv = [];
  let prev = null,
    bal = 0;
  for (const [t, c] of ev) {
    if (bal === 0 && prev !== null && t !== prev) itv.push([prev, t]);
    bal += c;
    prev = t;
  }
  return itv;
}
```
