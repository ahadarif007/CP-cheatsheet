# Basic Calculator

```js
function calculate(s) {
  s += '$';
  function evalAt(i) {
    const st = [];
    let sign = '+',
      num = 0;
    while (i < s.length) {
      const c = s[i];
      if (c === ' ') {
        i++;
        continue;
      }
      if (c >= '0' && c <= '9') {
        num = num * 10 + (c.charCodeAt(0) - 48);
        i++;
        continue;
      }
      if (c === '(') {
        const r = evalAt(i + 1);
        num = r[0];
        i = r[1];
      }
      if (!(c >= '0' && c <= '9')) {
        if (sign === '+') st.push(num);
        else if (sign === '-') st.push(-num);
        else if (sign === '*') st.push(st.pop() * num);
        else if (sign === '/') st.push((st.pop() / num) | 0);
        if (c === ')') return [st.reduce((a, b) => a + b, 0), i + 1];
        sign = c;
        num = 0;
        i++;
      }
    }
    return [st.reduce((a, b) => a + b, 0), i];
  }
  return evalAt(0)[0];
}
```
