# Basic Calculator

> Evaluate a math expression string with `+`, `-`, `*`, `/`, and parentheses.
> Uses a recursive descent parser. Appends `$` as a sentinel to signal end of input.

## How It Works

```
Input: "2*(3+4)-1"

evalAt(0): sees "2", then '*', then '('
  -> recurse evalAt(2): "3+4" -> stack [3, 4] -> sum = 7, return [7, index=5]
  -> stack = [2*7] = [14], then '-', then '1'
  -> stack = [14, -1] -> sum = 13
```

```js
function calculate(expression) {
  expression += '$';                           // sentinel to flush last number

  function evalAt(pos) {
    const operands = [];
    let sign = '+', num = 0;

    while (pos < expression.length) {
      const char = expression[pos];

      if (char === ' ') { pos++; continue; }

      if (char >= '0' && char <= '9') {
        num = num * 10 + (char.charCodeAt(0) - 48);  // build multi-digit number
        pos++;
        continue;
      }

      if (char === '(') {
        const [subResult, nextPos] = evalAt(pos + 1);  // recurse into parens
        num = subResult;
        pos = nextPos;
      }

      if (!(char >= '0' && char <= '9')) {
        // Apply the pending operator
        if (sign === '+') operands.push(num);
        else if (sign === '-') operands.push(-num);
        else if (sign === '*') operands.push(operands.pop() * num);
        else if (sign === '/') operands.push((operands.pop() / num) | 0);

        if (char === ')') {
          return [operands.reduce((a, b) => a + b, 0), pos + 1];
        }

        sign = char;
        num  = 0;
        pos++;
      }
    }
    return [operands.reduce((a, b) => a + b, 0), pos];
  }

  return evalAt(0)[0];
}
```
