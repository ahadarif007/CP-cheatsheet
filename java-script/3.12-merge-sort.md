# Merge Sort

> Merge sort divides the array in half, recursively sorts each half, then merges
> the two sorted halves. Always O(n log n) time, O(n) extra space.

## Algorithm Visualization

```
Split:    [38, 27, 43, 3, 9, 82, 10]
          [38, 27, 43]     [3, 9, 82, 10]
          [38] [27,43]     [3,9] [82,10]
          [38] [27][43]    [3][9] [82][10]

Merge:    [38] [27][43] -> [27,43] -> [27,38,43]
          [3][9] -> [3,9]  [82][10] -> [10,82] -> [3,9,10,82]

Final:    [27,38,43] + [3,9,10,82] -> [3,9,10,27,38,43,82]
```

```js
function mergeSort(arr) {
  if (arr.length <= 1) return arr;

  const mid   = arr.length >> 1;
  const left  = mergeSort(arr.slice(0, mid));
  const right = mergeSort(arr.slice(mid));

  return mergeSorted(left, right);
}

function mergeSorted(left, right) {
  let i = 0, j = 0;
  const merged = [];

  while (i < left.length && j < right.length) {
    merged.push(left[i] <= right[j] ? left[i++] : right[j++]);
  }
  while (i < left.length) merged.push(left[i++]);   // remaining from left
  while (j < right.length) merged.push(right[j++]); // remaining from right

  return merged;
}
```
