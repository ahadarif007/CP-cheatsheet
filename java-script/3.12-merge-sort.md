# Merge Sort

```js
function mergeSort(a) {
  if (a.length <= 1) return a;
  const m = a.length >> 1;
  const L = mergeSort(a.slice(0, m));
  const R = mergeSort(a.slice(m));
  return merge(L, R);
}
function merge(L, R) {
  let i = 0,
    j = 0;
  const res = [];
  while (i < L.length && j < R.length)
    res.push(L[i] <= R[j] ? L[i++] : R[j++]);
  while (i < L.length) res.push(L[i++]);
  while (j < R.length) res.push(R[j++]);
  return res;
}
```
