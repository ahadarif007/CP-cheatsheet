# Sliding Window

Longest substring without repeating

```js
function lengthOfLongestSubstring(s) {
  const last = Array(256).fill(-1);
  let best = 0,
    l = 0;
  for (let r = 0; r < s.length; r++) {
    const c = s.charCodeAt(r);
    if (last[c] >= l) l = last[c] + 1;
    last[c] = r;
    best = Math.max(best, r - l + 1);
  }
  return best;
}
```
